using System.Linq;
using Abp.Authorization;
using Abp.Configuration.Startup;
using Abp.Localization;
using Abp.MultiTenancy;
using GCT.Foundation;
using GCT.Foundation.Authorization;
using @{@data.Entity.Namespace};
using @data.ReplaceInfo.CompanyNamespace@{@data.ReplaceInfo.SolutionNamespace}.Authorization;

// ReSharper disable once CheckNamespace
namespace @data.ReplaceInfo.CompanyNamespace@{@data.ReplaceInfo.SolutionNamespace}.Authorization
{
    /// <summary>
    /// 权限配置都在这里。
    /// 给权限默认设置服务
    /// See <see cref="@{@data.Entity.Name}Permissions" /> 所有权限常量的定义位置 
    /// <see cref="@{@data.Entity.Name}"/> 实体位置
    ///前往 <see cref="@{@data.ReplaceInfo.SolutionNamespace}ApplicationModule"/>的AbpAutoMapper配置方法下添加以下代码段
    ///</summary>
    public class @{@data.Entity.Name}AuthorizationProvider : AuthorizationProvider
    {
        private readonly bool _isMultiTenancyEnabled;

		public @{@data.Entity.Name}AuthorizationProvider()
		{

		}

        /// <summary>
        /// 多租户配置
        /// </summary>
        /// <param name="isMultiTenancyEnabled"></param>       
        public @{@data.Entity.Name}AuthorizationProvider(bool isMultiTenancyEnabled)
        {
            _isMultiTenancyEnabled = isMultiTenancyEnabled;
        }

        public @{@data.Entity.Name}AuthorizationProvider(IMultiTenancyConfig multiTenancyConfig)
        {
            _isMultiTenancyEnabled = multiTenancyConfig.IsEnabled;
        }

        /// <summary>
        /// 在这里配置了@{@data.Entity.EntityDisplayName}的权限。
        /// </summary>
        /// <param name="context"></param>
		public override void SetPermissions(IPermissionDefinitionContext context)
		{
			// 在这里配置了@{@data.Entity.EntityDisplayName} 的权限。
			var pages = context.GetPermissionOrNull(AppPermissions.Pages) ?? context.CreatePermission(AppPermissions.Pages, L("Pages"));

			var administration = pages.Children.FirstOrDefault(p => p.Name == AppPermissions.Pages_Administration) ?? 
			pages.CreateChildPermission(AppPermissions.Pages_Administration, L("Administration"));

			var @{@data.Entity.LowerName} = administration.CreateChildPermission(@{@data.Entity.Name}Permissions.@{@data.Entity.Name}_Node, L("@{@data.Entity.Name}"));
			@{@data.Entity.LowerName}.CreateChildPermission(@{@data.Entity.Name}Permissions.@{@data.Entity.Name}_Query, L("Query"));
			@{@data.Entity.LowerName}.CreateChildPermission(@{@data.Entity.Name}Permissions.@{@data.Entity.Name}_Create, L("Create"));
			@{@data.Entity.LowerName}.CreateChildPermission(@{@data.Entity.Name}Permissions.@{@data.Entity.Name}_Edit, L("Edit"));
			@{@data.Entity.LowerName}.CreateChildPermission(@{@data.Entity.Name}Permissions.@{@data.Entity.Name}_Delete, L("Delete"));
            @{@data.Entity.LowerName}.CreateChildPermission(@{@data.Entity.Name}Permissions.@{@data.Entity.Name}_BatchDelete, L("BatchDelete"));
            @{@data.Entity.LowerName}.CreateChildPermission(@{@data.Entity.Name}Permissions.@{@data.Entity.Name}_Copy, L("Copy"));

			//// custom codes
					
			
			@{@ViewBag.OldCustomCode}
			//// custom codes end
		}
        /// <summary>
        /// 多语言本地化
        /// </summary>
        /// <param name="name"></param>
        /// <returns></returns>
		private static ILocalizableString L(string name)
		{
			return new LocalizableString(name, AppConsts.LocalizationSourceName);
		}
    }
}
