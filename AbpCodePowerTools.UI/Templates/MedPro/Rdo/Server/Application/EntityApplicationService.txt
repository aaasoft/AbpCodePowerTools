        /// <summary>
        ///     获取@{@data.Entity.EntityDisplayName}Rdo列表
        /// </summary>
        /// <param name="input"></param>
        /// <returns></returns>
        public async Task<PagedResultDto<RdoBaseDto>> GetRdoPaged(Get@{@data.Entity.Name}sInput input)
        {
            // @{@data.Entity.EntityDisplayName}base 查询
            var queryBase = _@{@data.Entity.LowerName}Manager.BaseQueryAsNoTracking
                .WhereIf(!input.FilterText.IsNullOrWhiteSpace(),
                    f => f.@{@data.Entity.Name}Name.Contains(input.FilterText));

            // 总数
            var rdoBaseTotalCount = await queryBase.CountAsync();

            // @{@data.Entity.EntityDisplayName}转换为标准的Rdo对象列表
            var rdoBaseList = await queryBase
                .Select(x => new RdoBaseDto
                {
                    Id = x.Id,
                    Name = x.@{@data.Entity.Name}Name,
                    RevOfRcdId = x.RevOfRcdId,
                    CreationTime = x.CreationTime
                })
                .OrderBy(input.Sorting)
                .PageBy(input)
                .ToListAsync();

            // @{@data.Entity.EntityDisplayName}base id集合
            var baseIds = rdoBaseList.Select(x => x.Id).ToList();

            // @{@data.Entity.EntityDisplayName}列表倒叙排列
            var rdoList = await _@{@data.Entity.LowerName}Manager.QueryAsNoTracking
                .Where(x => baseIds.Contains(x.@{@data.Entity.Name}BaseId))
                .Select(o => new RdoDto()
                {
                    Id = o.Id,
                    BaseId = o.@{@data.Entity.Name}BaseId,
                    Revision = o.@{@data.Entity.Name}Revision,
                    CreationTime = o.CreationTime
                })
                .ToListAsync();
            // @{@data.Entity.EntityDisplayName}列表字典化
            var rdoDict = rdoList.GroupBy(o => o.BaseId).ToDictionary(o => o.Key, o => o);

            // @{@data.Entity.EntityDisplayName}分组设置
            foreach (var rdo in rdoBaseList)
            {
                if (rdoDict.TryGetValue(rdo.Id, out var rdoDataList))
                {
                    rdo.RdoList = rdoDataList.AsQueryable()
                        .OrderByDescending(x => x.CreationTime)
                        .ToList();
                }
                else
                {
                    rdo.RdoList = new List<RdoDto>();
                }
            }

            var ret = new List<RdoBaseDto>();
            foreach (var baseDto in rdoBaseList)
            {
                if (baseDto.RdoList.Count > 0)
                {
                    ret.Add(baseDto);
                }
            }

            return new PagedResultDto<RdoBaseDto>(rdoBaseTotalCount, ret);
        }

        /// <summary>
        ///     获取@{@data.Entity.EntityDisplayName}Rdo列表for下拉列表
        /// </summary>
        /// <param name="input"></param>
        /// <returns></returns>
        [Microsoft.AspNetCore.Mvc.HttpPost]
        public async Task<List<RdoBaseDto>> GetRdoCombox(GetQueryFilterInput input)
        {
            //baseList
            var entityBaseList = await _@{@data.Entity.LowerName}Manager.BaseQueryAsNoTracking
                .WhereIf(!string.IsNullOrEmpty(input.FilterText), x => x.@{@data.Entity.Name}Name.Contains(input.FilterText))
                .Select(x => new RdoBaseDto
                {
                    Id = x.Id,
                    Name = x.@{@data.Entity.Name}Name,
                    RevOfRcdId = x.RevOfRcdId,
                    CreationTime = x.CreationTime
                })
                .OrderBy(input.Sorting)
                .ToListAsync();

            //实体base的id列表
            var baseIds = entityBaseList.Select(x => x.Id).ToList();

            // 传入 RdoId 为空，但 RdoBaseId 不为空，取默认版本
            if (!input.RdoId.HasValue && input.RdoBaseId.HasValue)
            {
                var baseDto = entityBaseList.FirstOrDefault(o => o.Id == input.RdoBaseId.Value);
                // 如果现有数据中没有查到base，那么就跳过
                if (baseDto != null)
                {
                    input.RdoId = baseDto.RevOfRcdId;
                }
            }

            //实体base下的@{@data.Entity.EntityDisplayName}列表
            var rdoListItems = await _@{@data.Entity.LowerName}Manager.QueryAsNoTracking
                .WhereIfThenElse(
                    input.RdoId.HasValue,
                    // then - 如果设置了当前选中 rdo 的 id
                    thenItem => thenItem.Id == input.RdoId.Value
                                || baseIds.Contains(thenItem.@{@data.Entity.Name}BaseId),
                    // else - 如果没有设置当前选中 rdo 的 id
                    elseItem => baseIds.Contains(elseItem.@{@data.Entity.Name}BaseId)
                )
                .Select(a => new RdoDto
                {
                    BaseId = a.@{@data.Entity.Name}BaseId,
                    Id = a.Id,
                    Revision = a.@{@data.Entity.Name}Revision,
                    CreationTime = a.CreationTime
                })
                .ToListAsync();

            // rdo base 的 rdo 集合
            foreach (var rdoBaseDto in entityBaseList)
            {
                rdoBaseDto.RdoList = rdoListItems
                    .Where(x => x.BaseId == rdoBaseDto.Id)
                    .OrderByDescending(o => o.CreationTime)
                    .ToList();
            }

            return entityBaseList.Where(o => o.RdoList != null && o.RdoList.Count > 0)
                .ToList();
        }

        {

        {
            var @{@data.Entity.LowerName}s = await this._@{@data.Entity.LowerName}Manager.GetRdoIncludeQuery()
                .Where(o => o.@{@data.Entity.Name}BaseId == input.Id)

            {
                @{@data.Entity.LowerName}Base = @{@data.Entity.LowerName}s.FirstOrDefault()?.@{@data.Entity.Name}Base;

            // 遍历删除
            foreach (var @{@data.Entity.LowerName} in @{@data.Entity.LowerName}s)

                // 创建追溯 - 删除
                await this._modelAuditHistoryManager.CreateOnDelete(@{@data.Entity.LowerName});
                // 删除doc
                await this._@{@data.Entity.LowerName}Manager.DeleteRdo(@{@data.Entity.LowerName});

            await _@{@data.Entity.LowerName}Manager.DeleteBase(@{@data.Entity.LowerName}Base);

        /// <summary>
        ///     复制@{@data.Entity.EntityDisplayName}
        /// </summary>
        /// <param name="input"></param>
        /// <returns></returns>
        [AbpAuthorize(@{@data.Entity.Name}Permissions.@{@data.Entity.Name}_Copy)]
        public async Task CopyRdo(CopyRdoDto input)
        {
            // 验证 base name 是否重复
            var existDuplicateName = await _@{@data.Entity.LowerName}Manager.BaseQueryAsNoTracking
                .Where(o => o.@{@data.Entity.Name}Name == input.Name)
                .AnyAsync();
            if (existDuplicateName)
            {
                RepetError(input.Name);
            }

            // 查找rdo数据
            var @{@data.Entity.LowerName} = await this._@{@data.Entity.LowerName}Manager.FindRdoByIdWithInclude(input.RdoId);

            // rdo 关联的数据
            var @{@data.Entity.LowerName}Base = @{@data.Entity.LowerName}.@{@data.Entity.Name}Base;

            // 新id
            var newId = _@{@data.Entity.LowerName}Manager.NewId();
            var newBaseId = _@{@data.Entity.LowerName}Manager.NewId();

            // rdo - 复制，指定新的参数
            @{@data.Entity.LowerName}.Id = newId;
            @{@data.Entity.LowerName}.@{@data.Entity.Name}BaseId = newBaseId;
            @{@data.Entity.LowerName}.@{@data.Entity.Name}Revision = input.Revision;
            // base - 复制，指定新的参数
            @{@data.Entity.LowerName}Base.Id = newBaseId;
            @{@data.Entity.LowerName}Base.@{@data.Entity.Name}Name = input.Name;
            @{@data.Entity.LowerName}Base.RevOfRcdId = newId;

            // 置空审计数据
            @{@data.Entity.LowerName}.EmptyAudit();
            @{@data.Entity.LowerName}Base.EmptyAudit();

            // 创建追溯 - 新增
            await this._modelAuditHistoryManager.CreateOnAdd(@{@data.Entity.LowerName});
            // 创建 rdo
            await this._@{@data.Entity.LowerName}Manager.CreateRdo(@{@data.Entity.LowerName});
            // 创建rdobase
            await this._@{@data.Entity.LowerName}Manager.CreateOrUpdateBase(@{@data.Entity.LowerName}Base);
        }

        /// <summary>
        ///     新增@{@data.Entity.EntityDisplayName}
        /// </summary>
        [AbpAuthorize(@{@data.Entity.Name}Permissions.@{@data.Entity.Name}_Create)]
        protected virtual async Task<RdoCreateOrUpdateOutput> Create(@{@data.Entity.Name}EditDto input)
        {
            // 置空参数
            input.Id = default;

            // 映射数据
            var entityBase = default(@{@data.Entity.Name}Base);
            // 从数据库查找 rdoBase
            if (input.@{@data.Entity.Name}BaseId.HasValue)
            {
                entityBase = await this._@{@data.Entity.LowerName}Manager.FindBaseById(input.@{@data.Entity.Name}BaseId.Value);
            }

            // 如果 rdoBase 不为空，那么将输入数据填充到 rdoBase 实例
            if (entityBase != null)
            {
                ObjectMapper.Map(input.@{@data.Entity.Name}Base, entityBase);
            }
            // 如果 rdoBase 为空,那么将输入数据映射为 rdoBase 实例
            else
            {
                entityBase = ObjectMapper.Map<@{@data.Entity.Name}Base>(input.@{@data.Entity.Name}Base);
                entityBase.Id = this._@{@data.Entity.LowerName}Manager.NewId();

                // 如果数据库没找到 rdoBase, 说明 rdoBase 也是新增的，因此rdo必然是默认版本
                input.IsRevOfRcdId = true;
            }


            // 将输入数据映射为 rdo 实例
            var entity = ObjectMapper.Map<@{@data.Entity.Name}>(input);

            // 关联 rdoBase
            entity.@{@data.Entity.Name}BaseId = entityBase.Id;
            entity.@{@data.Entity.Name}Base = entityBase;

            // 添加到数据库
            await _@{@data.Entity.LowerName}Manager.CreateOrUpdateRdoAndBase(entity, entityBase, input.IsRevOfRcdId);

            // 创建追溯 - 新增
            await _modelAuditHistoryManager.CreateOnAdd(entity);

            return new RdoCreateOrUpdateOutput()
            {
                Id = entity.Id,
                BaseId = entity.@{@data.Entity.Name}BaseId,
                Name = entityBase.@{@data.Entity.Name}Name,
                Version = entity.@{@data.Entity.Name}Revision,
                IsRevOfRcdId = entity.Id == entityBase.RevOfRcdId
            };

        }

        /// <summary>
        ///     编辑@{@data.Entity.EntityDisplayName}
        /// </summary>
        [AbpAuthorize(@{@data.Entity.Name}Permissions.@{@data.Entity.Name}_Edit)]
        protected virtual async Task<RdoCreateOrUpdateOutput> Update(@{@data.Entity.Name}EditDto input)
        {
            // 查找
            if (input.Id != null)
            {
                var entity = await this._@{@data.Entity.LowerName}Manager.FindRdoByIdWithInclude(input.Id.Value);
                // 记录修改之前的数据
                var beforeEntity = entity.JsonClone();

                // 映射@{@data.Entity.EntityDisplayName}和@{@data.Entity.EntityDisplayName}base
                ObjectMapper.Map(input, entity);
                ObjectMapper.Map(input.@{@data.Entity.Name}Base, entity.@{@data.Entity.Name}Base);

                // 提交到数据库
                entity =  await _@{@data.Entity.LowerName}Manager.CreateOrUpdateRdoAndBase(entity, entity.@{@data.Entity.Name}Base, input.IsRevOfRcdId);

                // 创建追溯 - 修改
                await _modelAuditHistoryManager.CreateOnUpdate(beforeEntity, entity);

                return new RdoCreateOrUpdateOutput()
                {
                    Id = entity.Id,
                    BaseId = entity.@{@data.Entity.Name}BaseId,
                    Name = entityBase.@{@data.Entity.Name}Name,
                    Version = entity.@{@data.Entity.Name}Revision,
                    IsRevOfRcdId = entity.Id == entityBase.RevOfRcdId
                };
            }

            return null;
        }
