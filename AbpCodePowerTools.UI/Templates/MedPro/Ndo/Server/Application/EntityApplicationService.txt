        {
            var @{@data.Entity.LowerName} = await _@{@data.Entity.LowerName}Manager.FindNdoByIdWithInclude(input.Id);

            if (@{@data.Entity.LowerName} == null)
            {
                NullError();

            var res = await _@{@data.Entity.LowerName}Manager
                .QueryAsNoTracking
                .FirstOrDefaultAsync(q => q.@{@data.Entity.Name}Name == input.Name.Trim());

            if (res != null)
            {
                RepetError(input.Name);
            }

            if (@{@data.Entity.LowerName} != null)
            {

                @{@data.Entity.LowerName}.EmptyAudit();

                @{@data.Entity.LowerName}.@{@data.Entity.Name}Name = input.Name;
                @{@data.Entity.LowerName}.Id = _@{@data.Entity.LowerName}Manager.NewId();

            }
