@model AbpDtoGenerator.GeneratorModels.ViewModel
@{
    var data = Model.SPAClient;
}


import { Component, Injector, OnInit, ViewChild } from '@@angular/core';
import { PagedRequestDto } from '@@commercial/l-52abp-ng';
import { GNdoPagedListingComponentBaseDirective } from '@@shared/component-base/g-ndo-paged-listing-component-base';
import { GConsts } from '@@shared/components/AppConsts/g-consts';
import { GCopyNdoComponent } from '@@shared/components/g-copy-ndo/g-copy-ndo.component';
import { NdoDto } from '@@shared/service-proxies/api/NdoDto';
import { finalize } from 'rxjs/operators';
import { FileDownloadService } from '@@shared/utils';
import { CreateOrEdit@{@data.Entity.Name}Component } from './create-or-edit-@{@data.Entity.LowerName}/create-or-edit-@{@data.Entity.LowerName}.component';
import { @{@data.Entity.Name}ServiceProxy } from '@@shared/service-proxies/api/@{@data.Entity.Name}ServiceProxy';

@@Component({
    templateUrl: './@{@data.Entity.SplitName}.component.html',
    styleUrls: ['./@{@data.Entity.SplitName}.component.less'],
    selector: 'app-@{@data.Entity.LowerName}',
})


export class  @{@data.Entity.Name}Component extends GNdoPagedListingComponentBaseDirective<NdoDto> implements OnInit {

  constructor(
    injector: Injector, 
    private _@{@data.Entity.LowerName}Service: @{@data.Entity.Name}ServiceProxy,
    private _fileDownloadService: FileDownloadService,) {
    super(injector, '@{@data.Entity.Namespace}.@{@data.Entity.Name}');
  }

  ngOnInit(): void {
    // super.ngOnInit() 中将会调用 getBtnList 和 refresh 方法
    super.ngOnInit();
  }


  /**
  * 获取后端数据列表信息
  * @@param request 请求的数据的dto 请求必需参数 skipCount: number; maxResultCount: number;
  * @@param pageNumber 当前页码
  * @@param finishedCallback 完成后回调函数，包含了loading效果关闭等操作
  */
  protected fetchDataList(request: PagedRequestDto, pageNumber: number, finishedCallback: () => void): void {
    this.loading = true;
    this._@{@data.Entity.LowerName}Service
      .getNdoPaged(this.filterText, request.sorting, request.maxResultCount, request.skipCount)
      .pipe(
        finalize(() => {
          finishedCallback();
          this.loading = false;
        }),
      )
      .subscribe((result) => {
        this.dataList = result.items;
        this.showPaging(result);
      });
  }

  /** 动态获取按钮组 */
  protected getBtnList() {
    this.btnList = [
      {
        icon: GConsts.gicon.create,
        text: GConsts.button.create,
        eventName: GConsts.button.create,
        display: this.isGranted('Pages.@{@data.Entity.Name}.Create'),
      },
      {
         icon: GConsts.gicon.delete,
        text: GConsts.button.delete,
        eventName: GConsts.button.delete,
        display: this.isGranted('Pages.@{@data.Entity.Name}.Delete') && this.id,
      },
      {
        icon: GConsts.gicon.refresh,
        text: GConsts.button.refresh,
        eventName: GConsts.button.refresh,
        display: this.isGranted('Pages.@{@data.Entity.Name}.Query') && this.isShowEdit,
        alignRight: true,
      },
      {
        // type: 'danger',
        icon: GConsts.gicon.gctzhuisu,
        text: GConsts.button.modelAudit,
        eventName: GConsts.button.modelAudit,
        display: !!this.id,
      },
      {
        type: 'primary',
        icon: GConsts.gicon.save,
        text: GConsts.button.save,
        eventName: GConsts.button.save,
        display: (this.isGranted('Pages.@{@data.Entity.Name}.Create') || this.isGranted('Pages.@{@data.Entity.Name}.Edit')) && this.isShowEdit,
        alignRight: true,
      },

      {
        icon: GConsts.gicon.export,
        text: GConsts.button.export,
        eventName: GConsts.button.export,
        display: this.isGranted('Pages.@{@data.Entity.Name}.Export'),
      },

      {
        icon: GConsts.gicon.import,
        text: GConsts.button.DownloadExcelTemplate,
        eventName: GConsts.button.DownloadExcelTemplate,
        display: this.isGranted('Pages.@{@data.Entity.Name}.Import'),
      },
      {
        icon: GConsts.gicon.import,
        text: GConsts.button.import,
        eventName: GConsts.button.import,
        display: this.isGranted('Pages.@{@data.Entity.Name}.Import'),
      },
      {
        icon: GConsts.gicon.copy,
        text: GConsts.button.copy,
        eventName: GConsts.button.copy,
        display: this.isGranted('Pages.@{@data.Entity.Name}.Copy') && this.id,
      },
    ];
  }


  
  /** 操作按钮点击事件 */
  btnCallback(eventName: string) {
    // 默认实现中已经处理了 create/delete/refresh/save/copy/modelAudit
  }
   

  //导出数据为Excel文件
  exportDataToExcel() {
    this._@{@data.Entity.LowerName}Service.get@{@data.Entity.Name}ToExcel().subscribe((result) => {
     
      this._fileDownloadService.downloadTempFile(result);
    });
  }

    //复制数据
  copy() {
    let data: any = {
      id: this.id,
      name: '@{@data.Entity.Name}Name',
    };
    this.loading = true;
    this._@{@data.Entity.LowerName}Service
        .getForEdit(this.id)
        .pipe(finalize(() => (this.loading = false)))
        .subscribe((res) => {
      data.value = res.@{@data.Entity.LowerName}.@{@data.Entity.LowerName}Name;
      this.modalHelper.createStatic(GCopyNdoComponent, { entity: data }).subscribe((result) => {
        if (result) {
        this.loading = true;
          this._@{@data.Entity.LowerName}Service.copyNdo(result)
            .pipe(finalize(() => (this.loading = false)))
            .subscribe((result) => {
            this.copySuccess(data.value);
            this.refreshGoFirstPage();
          });
        }
      });
    });
  }


  /** 删除功能 */
  delete(): void {
    abp.message.confirm(
      this.l('ConfirmDelete'), // 确认提示
      this.l('MessageWarn'), // 确认提示（可选参数）
      (isConfirmed) => {
        if (isConfirmed) {
          this.loading = true;
          this._@{@data.Entity.LowerName}Service
            .delete(this.id)
            .pipe(finalize(() => (this.loading = false)))
            .subscribe(() => {
              // 消息提示
              this.deleteSuccess();
              // 刷新
              this.refresh();
            });
        }
      },
    );
  }


  /*下载Excel模板文件*/
  downloadExcelTemplate() {
    this._@{@data.Entity.LowerName}Service.downloadExcelTemplate().subscribe((result) => {
      // 先导入这个service
      this._fileDownloadService.downloadTempFile(result);
    });
  }
  /*上传Excel文件准备导入*/
  importExcel(info: string) {
    this._@{@data.Entity.LowerName}Service.import@{@data.Entity.Name}(info).subscribe((item) => {
      this.gBtnGroupComponent.excelImportComponent.initImportReturn(item);
      this.refresh();
    });
  }


}
